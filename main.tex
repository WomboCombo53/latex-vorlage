% Hilfsmakro: setzt sectionname und richtet Kopf-/Fußzeile einheitlich ein.
% Verwendet \thepage, das automatisch die aktive Nummerierung (römisch/arabisch) nutzt.
\newcommand{\setupchapter}[1]{%
  \renewcommand{\sectionname}{#1}%
  \pagestyle{fancy}%
  \fancyhf{}%
  \fancyhead[L]{\sectionname}%
  \fancyhead[R]{\thepage}%
  \fancyfoot[L]{\creator}%
  \fancyfoot[R]{\company}%
  \renewcommand{\headrulewidth}{0.4pt}%
  \renewcommand{\footrulewidth}{0.4pt}%
}

% Hilfsmakro für Inhalts-/Abbildungs-/Tabellen-/Listingverzeichnis: setzt sectionname und
% definiert zusätzlich den "plain"-Pagestyle um, damit auch die (von LaTeX automatisch
% mit "plain" gestylte) erste Seite des jeweiligen Verzeichnisses den richtigen Kopf-/Fußzeileninhalt
% erhält. \pagestyle statt \thispagestyle sorgt dafür, dass auch mehrseitige Verzeichnisse
% auf Folgeseiten den korrekten Kopf behalten statt auf den Stil des vorherigen Kapitels zurückzufallen.
\newcommand{\setupverzeichnis}[1]{%
  \renewcommand{\sectionname}{#1}%
  \fancypagestyle{plain}{%
    \fancyhf{}%
    \fancyhead[L]{\sectionname}%
    \fancyhead[R]{\thepage}%
    \fancyfoot[L]{\creator}%
    \fancyfoot[R]{\company}%
    \renewcommand{\headrulewidth}{0.4pt}%
    \renewcommand{\footrulewidth}{0.4pt}%
  }%
  \pagestyle{plain}%
}
